    actor Staff as Staff/Admin
    participant FE as React Frontend
    participant BE as FastAPI Backend
    participant FaceEng as Face Engine (InsightFace)
    participant SQLite as SQLite DB
    participant Chroma as ChromaDB

    rect rgb(232, 245, 233)
        Note over Staff,SQLite: Login Flow
        Staff->>FE: Enter username & password
        FE->>BE: POST /api/v1/auth/login
        BE->>SQLite: Query staff by staff_code
        SQLite-->>BE: Staff record + hashed_password
        BE->>BE: Verify Argon2 hash
        BE-->>FE: JWT access_token
        FE->>FE: Store token in state
    end

    rect rgb(227, 242, 253)
        Note over Staff,Chroma: Face Registration Flow
        Staff->>FE: Upload student photo
        FE->>BE: POST /api/v1/recognition/register (multipart)
        BE->>FaceEng: Detect face & extract embedding
        FaceEng-->>BE: 512-dim embedding vector
        BE->>Chroma: Store embedding with student_id
        Chroma-->>BE: OK
        BE->>SQLite: UPDATE student SET face_data_registered=true
        BE-->>FE: Registration success
    end

    rect rgb(255, 243, 224)
        Note over Staff,Chroma: Attendance Session Flow
        Staff->>FE: Click "Start Attendance" for Class
        FE->>BE: POST /api/v1/attendance/start {group_id}
        BE->>SQLite: CREATE attendance_session (SCANNING)
        BE-->>FE: session_id
        
        loop Each Camera Frame
            FE->>BE: POST /api/v1/recognition/recognize (image)
            BE->>FaceEng: Detect faces in frame
            FaceEng-->>BE: Face embeddings[]
            BE->>Chroma: Search nearest match
            Chroma-->>BE: student_id + confidence
            BE->>SQLite: INSERT attendance_record (PRESENT, FACE)
            BE-->>FE: {student_name, confidence}
        end

        Staff->>FE: Click "Stop Attendance"
        FE->>BE: POST /api/v1/attendance/stop
        BE->>SQLite: UPDATE session SET status=COMPLETED
        BE-->>FE: Session summary
    end  dfd level o and 1 flowchart TB
